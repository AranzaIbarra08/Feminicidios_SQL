create table homicidas
(idH 	varchar(10) primary key,

create table crimenes
(idC varchar(20) primary key,

create table estado
(idE varchar(20) primary key,

create table inst
(idI varchar(10) primary key,

create table victimas
(idV varchar(20) primary key,
nombre_v 	varchar(30),
